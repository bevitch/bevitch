# Styles

## Background color

You can use following to set element color:

- use `.bv--color-(color)` (for example `.bv--color-red`) to set element background,
- use `.bv--color-(color)` and `.bv--darken-(1, 2 or 3)` (for example `.bv--darken-2`) to darken selected color,
- use `.bv--color-(color)` and `.bv--lighten-(1, 2 or 3)` (for example `.bv--lighten-2`) to lighten selected color,

<Preview>
  <Story name="Background color">
    {{
      data() {
        return {
          colors: bvColors
        }
      },
      methods: {
        getColor(prefix, color) {
          return `${prefix}${color}`;
        },
        getShade(prefix, shade) {
          const suffix = shade < 4 ? `lighten${shade - 4}` : shade === 4 ? '' : `darken-${shade - 4}`;
          return suffix.length > 0 ? `${prefix}${suffix}` : '';
        }
      },
      template: `
        <div>
          <div style="display: flex;flex-wrap: wrap;">
            <div v-for="(color, index) in colors" :key="index" style="width: 300px;">
              <div v-for="i in 7" :key="i" :class="[getColor('bv--color-', color), getShade('bv--', i)]">{{ getColor('.bv--color-', color) }} {{ getShade('.bv--', i) }}</div>
            </div>
          </div>
        </div>`,
    }}
  </Story>
</Preview>

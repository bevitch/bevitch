import { Meta, Props, Story, Preview } from "@storybook/addon-docs/blocks";
import { withA11y } from "@storybook/addon-a11y";
import { bvColors } from '@/config/storybook';
let colorMixin = {
  methods: {
    getBaseColor(prefix, color) {
      return `${prefix}${color}`;
    },
    getShade(shade) {
      const suffix = shade < 4 ? `lighten${shade - 4}` : shade === 4 ? '' : `darken-${shade - 4}`;
      return suffix.length > 0 ? `-${suffix}` : '';
    },
    getColor(prefix, color, shade){
      return `${this.getBaseColor(prefix, color)}${this.getShade(shade)}`;
    }
  }
};

# Styles

## Background color

You can use following to set element color:

- use `.bv--color-(color)` (for example `.bv--color-red`) to set element background color color,
- use `.bv--color-(color)-darken-(1, 2 or 3)` (for example `.bv--color-red-darken-2`) to set element background color to darken version of color,
- use `.bv--color-(color)-lighten-(1, 2 or 3)` (for example `.bv--color-red-lighten-2`) to set element background color to lighten version of color,

<Preview>
  <Story name="Background color">
    {{
      data() {
        return {
          colors: bvColors
        }
      },
      mixins: [
        colorMixin
      ],
      template: `
        <div>
          <div style="display: flex;flex-wrap: wrap;">
            <div v-for="(color, index) in colors" :key="index" style="width: 300px;">
              <div v-for="i in 7" :key="i" :class="getColor('bv--color-', color, i)">{{ getColor('.bv--color-', color, i) }}</div>
            </div>
          </div>
        </div>`,
    }}
  </Story>
</Preview>

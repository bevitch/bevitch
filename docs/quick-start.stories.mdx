# Quick start

## Adding library to application

### In your main file import `bevitch` library and inject it to your vue instance:

```ts
import Vue from 'vue';
import App from './App.vue';
import beVitch from '@bevitch/core';

Vue.use(beVitch);

new Vue({
  render: h => h(App)
}).$mount('#app');

```

### Add styles for library components:

```ts
import Vue from 'vue';
import App from './App.vue';
import beVitch from '@bevitch/core';
import '@bevitch/core/dist/lib/bevitch.css';

Vue.use(beVitch);

new Vue({
  render: h => h(App)
}).$mount('#app');
```

### Add your custom configuration:

```ts
import Vue from 'vue';
import App from './App.vue';
import beVitch from '@bevitch/core';
import '@bevitch/core/dist/lib/bevitch.css';

const beVitchOptions = {
  dictionary: {
    search: {
      search: 'Search...', 
      noMatchingItems: 'No matching items'
    }
  }
};

Vue.use(beVitch, beVitchOptions);

new Vue({
  render: h => h(App)
}).$mount('#app');
```

## TypeScript support

If you are using TypeScript you can import `CustomOptions` interface to validate your configuration:

```ts
import Vue from 'vue';
import App from './App.vue';
import beVitch from '@bevitch/core';
import '@bevitch/core/dist/lib/bevitch.css';
import { CustomOptions } from '@bevitch/core/src/types';

const beVitchOptions: CustomOptions = {
  dictionary: {
    search: {
      search: 'Search...', 
      noMatchingItems: 'No matching items'
    }
  }
};

Vue.use(beVitch, beVitchOptions);

new Vue({
  render: h => h(App)
}).$mount('#app');

```

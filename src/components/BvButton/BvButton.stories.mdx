# Button

<Props of={BvButton} />

Cofigure button compocent yourself to test all options:

<Preview>
  <Story name="Example">
    {{
      components: { BvButton },
      props: {
        light: {
          default: boolean('Light theme', false)
        },
        dark: {
          default: boolean('Dark theme', false)
        },
        color: {
          default: select('Color', bvColors, 'primary')
        },
        size: {
          default: select('Size', ['sm', 'md', 'lg'], 'md')
        },
        bordered: {
          default: boolean('Border', false)
        },
        elevated: {
          default: boolean('Elevated', false)
        },
        disabled: {
          default: boolean('Disabled', false)
        },
        rounded: {
          default: boolean('Rounded', false)
        },
        flat: {
          default: boolean('Flat', false)
        },
        outlined: {
          default: boolean('Outlined', false)
        },
        hoverText: {
          default: select('Hover text color for outlined and flat button', bvColors, 'white')
        },
      },
      template: `
        <BvButton
          :dark="dark"
          :color="color"
          :size="size"
          :bordered="bordered"
          :elevated="elevated"
          :disabled="disabled"
          :rounded="rounded"
          :outlined="outlined"
          :hoverText="hoverText"
        >
          Configure me
        </BvButton>
      `
    }}
  </Story>
</Preview>

<Preview>
  <Story name="Example">
    {{
      components: { BvButton },
      props: {
        dark: {
          default: boolean('Dark', false)
        },
        color: {
          default: select('Color', [
            'default',
            'primary',
            'secondary',
            'success',
            'info',
            'warning',
            'error',
            'red',
            'green-light'
          ], 'default')
        },
        size: {
          default: select('Size', ['sm', 'md', 'lg'], 'md')
        },
        bordered: {
          default: boolean('Border', false)
        },
        elevated: {
          default: boolean('Elevated', false)
        },
        disabled: {
          default: boolean('Disabled', false)
        },
        rounded: {
          default: boolean('Rounded', false)
        },
        outlined: {
          default: boolean('Outlined', false)
        },
      },
      template: `
        <BvButton
          :dark="dark"
          :color="color"
          :size="size"
          :bordered="bordered"
          :elevated="elevated"
          :disabled="disabled"
          :rounded="rounded"
          :outlined="outlined"
        >
          Configure me
        </BvButton>
      `
    }}
  </Story>
</Preview>
